function _generateFingerprint(){
  try{
    // Try Capacitor Device plugin (Android native UUID)
    if(window.Capacitor&&window.Capacitor.isNativePlatform&&window.Capacitor.Plugins&&window.Capacitor.Plugins.Device){
      // async but we try sync first; if not available, fall through
    }
  }catch(e){}
  // Web fingerprint from stable browser properties
  var parts=[];
  parts.push(screen.width+"x"+screen.height);
  parts.push(screen.colorDepth);
  parts.push(navigator.language||"");
  parts.push(navigator.hardwareConcurrency||"");
  parts.push(navigator.platform||"");
  var canvas=document.createElement("canvas");
  canvas.width=200;canvas.height=50;
  try{
    var ctx=canvas.getContext("2d");
    ctx.textBaseline="top";ctx.font="14px Arial";
    ctx.fillStyle="#f60";ctx.fillRect(125,1,62,20);
    ctx.fillStyle="#069";ctx.fillText("DeviceID:"+navigator.userAgent,2,15);
    ctx.fillStyle="rgba(102,204,0,0.7)";ctx.fillText("DeviceID:"+navigator.userAgent,4,17);
    parts.push(canvas.toDataURL().slice(-50));
  }catch(e){parts.push("no-canvas");}
  var str=parts.join("||");
  var hash=0;
  for(var i=0;i<str.length;i++){hash=((hash<<5)-hash)+str.charCodeAt(i);hash|=0;}
  return "D-"+Math.abs(hash).toString(36).toUpperCase();
}

// ── GPS → Division/Region/District/Upazila auto-select ──
// Nominatim already returns structured admin names in its response; the
// old code discarded everything except a flattened address string. This
// matches those names against our own BD/BD_UPAZILA/BD_DIVISIONS lists so
// the four cascading dropdowns fill themselves instead of the SAAO tapping
// through each one by hand.
function _norm(s){return (s||"").toString().replace(/\s*(district|zila|division|উপজেলা|জেলা|বিভাগ)\s*/gi,"").trim().toLowerCase()}
function _findKeyMatch(candidates,keys){
  for(var i=0;i<candidates.length;i++){
    var c=_norm(candidates[i]); if(!c)continue;
    for(var j=0;j<keys.length;j++){ if(_norm(keys[j])===c) return keys[j]; }
  }
  // fallback: substring match
  for(var i=0;i<candidates.length;i++){
    var c=_norm(candidates[i]); if(!c)continue;
    for(var j=0;j<keys.length;j++){ var k=_norm(keys[j]); if(k&&(c.indexOf(k)>=0||k.indexOf(c)>=0)) return keys[j]; }
  }
  return null;
}
window.autoSelectAdminFromGeo=function(addr){
  if(!addr) return;
  try{
    // Division (Nominatim "state" is typically "X Division" / "X বিভাগ")
    var divMatch=_findKeyMatch([addr.state],BD_DIVISIONS);
    if(divMatch){document.getElementById("fDivision").value=divMatch;}

    // District → also tells us which DAE Region contains it
    var districtCandidates=[addr.state_district,addr.district,addr.city,addr.county];
    var distMatch=_findKeyMatch(districtCandidates,Object.keys(BD_UPAZILA));
    if(distMatch){
      var regionMatch=null;
      Object.keys(BD).forEach(function(r){ if(BD[r].indexOf(distMatch)>=0) regionMatch=r; });
      if(regionMatch){
        document.getElementById("fRegion").value=regionMatch;
        onFormRegionChange(); // repopulates fDistrict for that region
        document.getElementById("fDistrict").value=distMatch;
        onFormDistrictChange(); // repopulates fUpazila for that district

        // Upazila (Nominatim "county"/"subdistrict" often holds this)
        var upaMatch=_findKeyMatch([addr.county,addr.subdistrict,addr.suburb],BD_UPAZILA[distMatch]||[]);
        if(upaMatch) document.getElementById("fUpazila").value=upaMatch;
      }
    }
  }catch(e){console.error("Auto admin-select failed:",e)}
};

function addCustomUpazila(district,name,skipSync){
  var obj={}; try{obj=JSON.parse(localStorage.getItem(CUSTOM_UPAZILA_KEY)||"{}")}catch(e){}
  obj[district]=obj[district]||[];
  if(obj[district].indexOf(name)<0) obj[district].push(name);
  localStorage.setItem(CUSTOM_UPAZILA_KEY,JSON.stringify(obj));
  // Sync to the sheet so every officer benefits from one person's addition,
  // not just this device. Best-effort: a failed sync still leaves the name
  // usable locally (matches this app's offline-first pattern elsewhere).
  if(!skipSync){
    var profile=getProfile()||{};
    fetch(GAS_SYNC_ENDPOINT,{method:"POST",headers:{"Content-Type":"application/json"},body:JSON.stringify({
      entryType:"custom_upazila", district:district, upazilaName:name,
      addedBy:profile.name||"", deviceId:getDeviceId()

  }

    _directoryPeople=cached.people;
    populateDirectoryDatalists();
    var ageMs=Date.now()-new Date(cached.cachedAt).getTime();
    if(ageMs<DIRECTORY_CACHE_MAX_AGE_MS) return;
  }
  fetch(GAS_SYNC_ENDPOINT+"?directory=1")
    .then(function(res){return res.json()})
    .then(function(data){
      if(!data||!data.ok||!Array.isArray(data.people)) return;
      _directoryPeople=data.people;
      saveDirectoryCache(data.people);
      populateDirectoryDatalists();
    })
    .catch(function(err){console.warn("Directory fetch failed (using cache if any):",err)});
}
// When the typed name exactly matches a known person, auto-fill their
// mobile number — saves re-typing a number that's already on record.
window.onDirectoryNameInput=function(kind,name){
  if(!_directoryPeople) return;
  var match=_directoryPeople.find(function(p){return p.name===name});
  if(!match) return;
  var mobileField=document.getElementById(kind==="saao"?"fSaaoMobile":"fOfficerMobile");
  if(mobileField&&!mobileField.value) mobileField.value=match.mobile;
};
