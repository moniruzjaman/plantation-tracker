async function sendToGAS(rows){
  try {
    var res = await fetch(GAS_SYNC_ENDPOINT, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(rows)
    });
    if(!res.ok){
      return { ok:false, error: "HTTP " + res.status };
    }
    return await res.json();
  } catch(e){
    return { ok:false, error: e.message };
  }
}

window.reverseGeocode = function(lat, lng, targetId) {
  var addrField = document.getElementById(targetId);
  if (!addrField) return;
  addrField.placeholder = "🌐 জিপিএস থেকে ঠিকানা লোড হচ্ছে...";
  
  var url = "https://nominatim.openstreetmap.org/reverse?format=jsonv2&lat=" + lat + "&lon=" + lng + "&accept-language=bn,en";
  fetch(url, {
    headers: {
      "User-Agent": "PlantationTracker/1.0"
    }
  })
  .then(function(res) { return res.json(); })
  .then(function(data) {
    if (data && data.display_name) {
      var addressStr = data.display_name;
      if (data.address) {
        var parts = [];
        var a = data.address;
        
        // Match standard local/regional divisions
        var village = a.village || a.suburb || a.neighbourhood || a.hamlet || a.road || "";
        var upazilaLoc = a.county || a.subdistrict || "";
        var districtLoc = a.state_district || a.district || a.city || "";
        var postOffice = a.postcode ? "পোস্টকোড: " + a.postcode : "";
        
        if (village) parts.push(village);
        if (upazilaLoc) parts.push(upazilaLoc);
        if (districtLoc) parts.push(districtLoc);
        if (postOffice) parts.push(postOffice);
        
        if (parts.length > 0) {
          addressStr = parts.join(", ");
        }
        if (targetId === "fAddress" && window.autoSelectAdminFromGeo) {
          window.autoSelectAdminFromGeo(a);
        }
      }
      addrField.value = addressStr;
    }
  })
  .catch(function(err) {
    console.error("Geocoding failed:", err);
  });
};

    var lat = parseFloat(m[1]).toFixed(6);
    var lng = parseFloat(m[2]).toFixed(6);

  }
};

// ── NDVI auto-fill ──
window.autoFillNdvi = function(lat, lng){
  lat = lat || (_val("fGeo").split(",")[0] || "").trim();
  lng = lng || (_val("fGeo").split(",")[1] || "").trim();
  if(!lat || !lng) return;
  var ndviEl = document.getElementById("fNdvi");
  if(!ndviEl) return;
  ndviEl.value = "গণনা হচ্ছে...";
  var dLat = 0.005, dLng = 0.005;
  fetch("/api/gee-proxy", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      bounds: { west: parseFloat(lng)-dLng, south: parseFloat(lat)-dLat, east: parseFloat(lng)+dLng, north: parseFloat(lat)+dLat },
      year: new Date().getFullYear()
    })
  })
  .then(function(r){ return r.json(); })
  .then(function(data){
    if(data && typeof data.ndvi === "number"){
      ndviEl.value = data.ndvi.toFixed(3);
    } else {
      ndviEl.value = "";
    }
  })
  .catch(function(){
    ndviEl.value = "";
  });
};

// countSeedlings — supports BOTH the new flat `seedlings` array shape and
// the legacy fruitSeedlings/forestSeedlings/medicinalSeedlings shape.
function countSeedlings(s){
  var r={fruit:0,forest:0,medicinal:0};
  if(s && Array.isArray(s.seedlings)){
    s.seedlings.forEach(function(e){
      var qty=parseInt(e.quantity)||0;
      var cat=(e.category||"").trim();
      if(cat.indexOf("ফল")===0||cat==="fruit") r.fruit+=qty;
      else if(cat.indexOf("বন")===0||cat==="forest") r.forest+=qty;
      else if(cat.indexOf("ঔষ")===0||cat==="medicinal") r.medicinal+=qty;
      else r.fruit+=qty;
    });
    return r;
  }
  ["fruitSeedlings","forestSeedlings","medicinalSeedlings"].forEach(function(k,i){
    (s[k]||[]).forEach(function(e){
      r[["fruit","forest","medicinal"][i]]+=(parseInt(e.count)||0)+(parseInt(e.graftingCount)||0);
    });
  });
  return r;
}

function populateMyFilterDropdowns(subs){
  var srcSel=document.getElementById("myFilterSource");
  var distSel=document.getElementById("myFilterDistrict");
  var upzSel=document.getElementById("myFilterUpazila");
  if(!srcSel||!distSel||!upzSel) return;
  var sources={},districts={},upazilas={};
  subs.forEach(function(s){
    if(s.sourceType) sources[s.sourceType]=1;
    if(s.district) districts[s.district]=1;
    if(s.upazila) upazilas[s.upazila]=1;
  });
  var keepSrc=srcSel.value,keepDist=distSel.value,keepUpz=upzSel.value;
  var mkOpts=function(obj,allLabel){
    return '<option value="">'+allLabel+'</option>'+Object.keys(obj).sort().map(function(v){return '<option value="'+v+'">'+v+'</option>'}).join("");
  };
  srcSel.innerHTML=mkOpts(sources,"সব উৎস");
  distSel.innerHTML=mkOpts(districts,"সব জেলা");
  upzSel.innerHTML=mkOpts(upazilas,"সব উপজেলা");
  srcSel.value=keepSrc;distSel.value=keepDist;upzSel.value=keepUpz;
}

};
