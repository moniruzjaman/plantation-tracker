// Bangladesh's real extent, with a generous buffer, used to reject clearly
// bad/placeholder GPS coordinates (e.g. "0,0" null-island, a GPS glitch, or
// a lat/lng typo) before they can ever reach fitBounds(). A single such
// point mixed in with real Bangladesh coordinates forces Leaflet to zoom
// out to fit both — which is exactly what a "world map" view looks like.
var BD_BOUNDS = { minLat: 20.0, maxLat: 27.5, minLng: 87.5, maxLng: 93.5 };
function isValidBdCoord(lat, lng) {
  if (isNaN(lat) || isNaN(lng)) return false;
  if (lat === 0 && lng === 0) return false; // null island
  return lat >= BD_BOUNDS.minLat && lat <= BD_BOUNDS.maxLat && lng >= BD_BOUNDS.minLng && lng <= BD_BOUNDS.maxLng;
}

// ── National App_Entry sync for the Map tab ──
// The Map should show every location recorded in the App_Entry sheet
// (columns L=অক্ষাংশ/latitude, M=দ্রাঘিমাংশ/longitude), across every
// officer's device — not just this browser's localStorage. The GAS proxy
// (/api/gas-sync?list=1) fetches and groups the sheet rows server-side
// (one object per অ্যাপ জমা আইডি, with a seedlings[] array), in the same
// shape produced locally by buildBaseSubmission()/toGASRows(), so the
// existing marker-rendering code below needs no special-casing.
var _nationalEntries = null;        // cache: array | null (not loaded yet)
var _nationalEntriesAt = null;      // Date of last successful fetch
var _nationalEntriesLoading = false;
var NATIONAL_ENTRIES_CACHE_KEY = "national_entries_cache_v1";
var NATIONAL_ENTRIES_CACHE_MAX_AGE_MS = 30 * 60 * 1000; // 30 minutes

function loadCachedNationalEntries() {
  try {
    var raw = localStorage.getItem(NATIONAL_ENTRIES_CACHE_KEY);
    if (!raw) return null;
    var parsed = JSON.parse(raw);
    if (!parsed || !Array.isArray(parsed.entries)) return null;
    return parsed;
  } catch (e) { return null; }
}
function saveCachedNationalEntries(entries) {
  try {
    localStorage.setItem(NATIONAL_ENTRIES_CACHE_KEY, JSON.stringify({ entries: entries, cachedAt: new Date().toISOString() }));
  } catch (e) { /* storage full/unavailable — ignore, cache is best-effort */ }
}

function fetchNationalEntries(force) {
  if (_nationalEntriesLoading) return Promise.resolve(_nationalEntries || []);
  if (_nationalEntries && !force) return Promise.resolve(_nationalEntries);

  // Serve the persisted cache immediately (if any) so the map isn't empty
  // while the network request is in flight; a fresh fetch still runs unless
  // the cache is very recent.
  var cached = !force ? loadCachedNationalEntries() : null;
  if (cached && !_nationalEntries) {
    _nationalEntries = cached.entries;
    _nationalEntriesAt = new Date(cached.cachedAt);
    var cacheAgeMs = Date.now() - _nationalEntriesAt.getTime();
    var statusElInit = document.getElementById("mapSyncStatus");
    if (statusElInit) {
      statusElInit.textContent = "📦 ক্যাশ থেকে " + _nationalEntries.length + " টি এন্ট্রি (" + _nationalEntriesAt.toLocaleTimeString("bn-BD") + ") — নতুন তথ্য আনা হচ্ছে...";
    }
    if (cacheAgeMs < NATIONAL_ENTRIES_CACHE_MAX_AGE_MS) {
      // Still kick off a background refresh, but don't block on it.
      fetchNationalEntries(true).then(function() { if (typeof renderMap === "function") renderMap(); });
      return Promise.resolve(_nationalEntries);
    }
  }

  _nationalEntriesLoading = true;
  var statusEl = document.getElementById("mapSyncStatus");
  if (statusEl) statusEl.textContent = "📡 App_Entry (সকল জেলা) থেকে তথ্য লোড হচ্ছে...";
  return fetch(GAS_SYNC_ENDPOINT + "?list=1")
    .then(function(res) { return res.json(); })
    .then(function(data) {
      var rows = (data && data.ok && Array.isArray(data.entries)) ? data.entries : [];
      _nationalEntries = rows.map(function(r) { r._source = "national"; return r; });
      _nationalEntriesAt = new Date();
      saveCachedNationalEntries(_nationalEntries);
      if (statusEl) {
        statusEl.textContent = "✅ App_Entry থেকে " + _nationalEntries.length + " টি এন্ট্রি সিঙ্ক করা হয়েছে (" + _nationalEntriesAt.toLocaleTimeString("bn-BD") + ")";
      }
      return _nationalEntries;
    })
    .catch(function(err) {
      console.error("National App_Entry fetch failed:", err);
      if (!_nationalEntries) {
        var fallback = loadCachedNationalEntries();
        _nationalEntries = fallback ? fallback.entries : [];
      }
      if (statusEl) statusEl.textContent = "⚠️ App_Entry থেকে তথ্য আনা যায়নি — ক্যাশ/এই ডিভাইসের তথ্য দেখানো হচ্ছে।";
      return _nationalEntries;
    })
    .finally(function() { _nationalEntriesLoading = false; });
}

// Combine this device's local (possibly not-yet-synced) submissions with
// the national App_Entry list, de-duplicated by অ্যাপ জমা আইডি
// (submissionId) — preferring the national (sheet-confirmed) copy — then
// apply the same region/district filters as the rest of the Map tab.
function getMapEntries() {
  var r = document.getElementById("mapRegion") && document.getElementById("mapRegion").value;
  var d = document.getElementById("mapDistrict") && document.getElementById("mapDistrict").value;
  var u = document.getElementById("mapUpazila") && document.getElementById("mapUpazila").value;

  var byId = {};
  var merged = [];

  (_nationalEntries || []).forEach(function(s) {
    var key = s.submissionId || s.id;
    if (key) byId[key] = true;
    merged.push(s);
  });
  getSubmissions().forEach(function(s) {
    var key = s.submissionId || s.id;
    if (key && byId[key]) return; // already present via the national (synced) copy
    s._source = s._source || "local";
    merged.push(s);
  });

  return merged.filter(function(s) {
    if (r && s.region !== r) return false;
    if (d && s.district !== d) return false;
    if (u && s.upazila !== u) return false;
    return true;
  });
}

window.renderMap = function() {
  try {
    var mapContainer = document.getElementById("leafletMap");
    if (!mapContainer) return;
    
    // Initialize map if not done already
    if (!mapInstance) {
      // Default viewpoint focused on Bangladesh
      mapInstance = L.map('leafletMap').setView([23.8103, 90.4125], 7);

      var osmLayer = L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
        maxZoom: 19,
        attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a>'
      }).addTo(mapInstance);

      var satelliteLayer = L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', {
        maxZoom: 19,
        attribution: 'Tiles &copy; Esri'
      });

      // NDVI / EVI: NASA GIBS MODIS 8-day vegetation index composites (free, no API key).
      // Composites lag a few days behind real-time, so we request a date ~6 days back
      // and silently fall back to OSM if a tile set turns out unavailable.
      var gibsDate = new Date(Date.now() - 6 * 24 * 60 * 60 * 1000).toISOString().slice(0, 10);
      var ndviLayer = L.tileLayer(
        'https://gibs.earthdata.nasa.gov/wmts/epsg3857/best/MODIS_Terra_NDVI_8Day/default/' + gibsDate + '/GoogleMapsCompatible_Level9/{z}/{y}/{x}.png',
        { maxZoom: 9, attribution: 'NASA GIBS — MODIS Terra NDVI (8-Day)', errorTileUrl: '' }
      );
      var eviLayer = L.tileLayer(
        'https://gibs.earthdata.nasa.gov/wmts/epsg3857/best/MODIS_Terra_EVI_8Day/default/' + gibsDate + '/GoogleMapsCompatible_Level9/{z}/{y}/{x}.png',
        { maxZoom: 9, attribution: 'NASA GIBS — MODIS Terra EVI (8-Day)', errorTileUrl: '' }
      );
      [ndviLayer, eviLayer].forEach(function(l) {
        var tileErrCount = 0;
        l.on('tileerror', function() {
          tileErrCount++;
          if (tileErrCount > 3 && mapInstance.hasLayer(l)) {
            mapInstance.removeLayer(l);
            mapInstance.addLayer(osmLayer);
          }
        });
      });

      L.control.layers({
        '🗺️ রাস্তার মানচিত্র': osmLayer,
        '🛰️ স্যাটেলাইট': satelliteLayer,
        '🌿 NDVI (উদ্ভিদ সূচক)': ndviLayer,
        '🍃 EVI (বর্ধিত উদ্ভিদ সূচক)': eviLayer
      }, null, { collapsed: true, position: 'topright' }).addTo(mapInstance);
    }
    
    // Clear old markers
    mapMarkers.forEach(function(m) { mapInstance.removeLayer(m); });
    mapMarkers = [];
    
    // Kick off (or reuse the cached) national App_Entry fetch. First paint
    // uses whatever is cached/local; once the fetch resolves we re-render
    // automatically so markers appear without the user having to act.
    if (!_nationalEntries && !_nationalEntriesLoading) {
      fetchNationalEntries().then(function() { renderMap(); });
    }
    
    // Get filtered submissions (local + national App_Entry, deduped, district/region/upazila filtered)
    var subs = getMapEntries();
    var _mapUpazilaLegend = {}; // rebuilt each render for the dynamic legend
    
    var bounds = [];
    var mappedCount = 0;
    var totalCount = subs.length;
    
    subs.forEach(function(s) {
      var geoStr = (s.geoLocation || "").trim();
      if (!geoStr) return;
      var parts = geoStr.split(",");
      if (parts.length !== 2) return;
      var lat = parseFloat(parts[0]);
      var lng = parseFloat(parts[1]);
      if (!isValidBdCoord(lat, lng)) {
        if (!isNaN(lat) && !isNaN(lng)) console.warn("Skipping out-of-range/placeholder coordinate for map marker:", s.id || s.submissionId, lat, lng);
        return;
      }
      
      bounds.push([lat, lng]);
      mappedCount++;
      
      var t = countSeedlings(s);
      var totalScCount = t.fruit + t.forest + t.medicinal;
      
      // Seedlings detailed details
      var _nm = s.farmerName || s.nurseryName || "";
      var _mb = s.farmerMobile || s.mobile || "";
      var _off = s.officerName || s.caretakerName || "";
      var _offmb = s.officerMobile || s.caretakerMobile || "";
      var _reg = s.region || s.division || "";
      var _sid = s.id || s.submissionId || "";
      var detailsHtml = '<div style="font-family: inherit; font-size:12px; line-height:1.5; min-width: 200px;">';
      detailsHtml += '<p style="font-weight:700; color:#15803d; font-size:13px; margin:0 0 4px 0;">' + _nm + '</p>';
      if(_mb){detailsHtml += '<p style="margin:2px 0;"><b>মোবাইল:</b> <a href="tel:' + _mb + '" style="color:#2563eb; font-weight:600; text-decoration:underline;">' + _mb + '</a></p>';}
      if(_off){detailsHtml += '<p style="margin:2px 0;"><b>মনিটরিং অফিসার:</b> ' + _off + (_offmb ? ' (<a href="tel:' + _offmb + '" style="color:#2563eb; text-decoration:underline;">' + _offmb + '</a>)' : '') + '</p>';}
      var pD = s.plantingDate ? new Date(s.plantingDate).toLocaleDateString("bn-BD") : (s.submittedAt ? new Date(s.submittedAt).toLocaleDateString("bn-BD") : "");
      if (pD) {
        detailsHtml += '<p style="margin:2px 0;"><b>রোপণের তারিখ:</b> ' + pD + '</p>';
      }
      detailsHtml += '<p style="margin:2px 0;"><b>রোপণকৃত বৃক্ষের অবস্থান:</b> ' + (s.address || "উল্লেখ নেই") + '</p>';
      detailsHtml += '<p style="font-size:10px; color:#6b7280; margin:2px 0;">(' + _reg + ' > ' + s.district + ' > ' + s.upazila + ')</p>';
      detailsHtml += '<div style="display:grid; grid-template-columns: repeat(3, 1fr); gap: 4px; text-align: center; font-size:10px; margin-top:6px;">';
      detailsHtml += '<div style="background:#fff7ed; border-radius:4px; padding:3px;"><span style="color:#ea580c; font-weight:bold;">' + t.fruit + '</span><br><span style="color:#6b7280;">ফলদ</span></div>';

      detailsHtml += '<div style="background:#eff6ff; border-radius:4px; padding:3px;"><span style="color:#1d4ed8; font-weight:bold;">' + t.medicinal + '</span><br><span style="color:#6b7280;">ঔষধি</span></div>';
      detailsHtml += '</div>';
      detailsHtml += '<button onclick="parent.window.editMySubmission ? parent.window.editMySubmission(\''+_sid+'\') : editMySubmission(\''+_sid+'\')" style="width:100%; margin-top:8px; background:#dbeafe; color:#1d4ed8; border:none; padding:5px; border-radius:6px; font-weight:700; cursor:pointer; font-size:11px;">✏️ তথ্য সম্পাদনা</button>';
      detailsHtml += '</div>';
      
      // Marker color logic: each upazila gets a distinct color from the
      // palette — this makes it easy to visually cluster by upazila on the map.
      // Special overrides: local-unsynced = gray, current user's own entry = purple.
      var markerBg = upazilaColor(s.upazila);
      if (s._source === "local") markerBg = '#9ca3af'; // Gray — not yet synced
      var myProfile = getProfile();
      if (!s._isMine) {
        var pm = myProfile && myProfile.mobile;
        if (pm && (s.saaoMobile === pm || s.officerMobile === pm)) s._isMine = true;
      }
      if (s._source === "local" || s._isMine) markerBg = '#a855f7'; // Purple — my entry
      
      // Track upazila→color for the dynamic legend
      if (s.upazila && s._source !== "local" && !s._isMine) {
        _mapUpazilaLegend[s.upazila] = markerBg;
      }

      var customIcon = L.divIcon({
        html: '<div style="background-color: ' + markerBg + '; width: 28px; height: 28px; border-radius: 50%; border: 2px solid white; display: flex; align-items: center; justify-content: center; color: white; font-size: 14px; box-shadow: 0 2px 6px rgba(0,0,0,0.3); transform: scale(1);" title="' + (s.upazila||'') + '">🌳</div>',

        iconSize: [28, 28],
        iconAnchor: [14, 14]
      });
      
      var marker = L.marker([lat, lng], { icon: customIcon }).addTo(mapInstance).bindPopup(detailsHtml);
      mapMarkers.push(marker);
    });
    
    // Build dynamic upazila color legend
    var legendEl = document.getElementById("mapUpazilaLegend");
    if (legendEl) {
      var keys = Object.keys(_mapUpazilaLegend).sort();
      if (keys.length > 0) {
        legendEl.innerHTML = keys.map(function(u) {
          return '<span class="inline-flex items-center gap-1 ml-3"><span class="w-3 h-3 rounded-full inline-block" style="background:' + _mapUpazilaLegend[u] + '"></span>' + u + '</span>';
        }).join('');
        legendEl.classList.remove('hidden');
      } else {
        legendEl.classList.add('hidden');
      }
    }

    // Update footer text statistics
    var statsText = document.getElementById("mapStatsText");
    if (statsText) {
      statsText.textContent = "মোট রোপণকারী: " + totalCount + " | ম্যাপে প্রদর্শিত: " + mappedCount;
    }
    
    // Zoom/fit map bounds
    if (bounds.length > 0) {
      mapInstance.fitBounds(bounds, { padding: [50, 50], maxZoom: 16 });
    } else {
      // No entries yet (e.g. still loading, or filtered to zero) — stay on
      // the national Bangladesh-focused view rather than jumping to any
      // single district, so the map always opens showing the whole country.
      mapInstance.setView([23.8103, 90.4125], 7);
    }
    
    // Correct Leaflet display inside hidden container — the map may have
    // been created (or last measured) while the tab was display:none, which
    // leaves Leaflet's internal size/zoom projection wrong (it can render as
    // a zoomed-way-out "world map" with markers in the wrong place). Fixing
    // the size alone isn't enough — the view must be re-applied afterward.
    setTimeout(function() {
      mapInstance.invalidateSize({ animate: false });
      if (bounds.length > 0) {
        mapInstance.fitBounds(bounds, { padding: [50, 50], maxZoom: 16 });
      } else {
        mapInstance.setView([23.8103, 90.4125], 7);
      }
    }, 250);
    
  } catch (err) {
    console.error("Map rendering error:", err);
  }
};

      deferredPrompt = null;
      var installBanner = document.getElementById('pwaInstallBanner');
      if (installBanner) installBanner.classList.add('hidden');
    });
  } else {
    alert("আপনার ফোন বা ব্রাউজার থেকে সরাসরি অ্যাপ হিসেবে ইনস্টল করতে ব্রাউজারের থ্রি-ডট (triple-dot) বা শেয়ার মেনুতে গিয়ে 'Add to Home Screen' অথবা 'Install' এ ক্লিক করুন।");
  }
};

function checkIframeAndPrompt() {
  var isLinkedIframe = false;
  try {
    isLinkedIframe = window.self !== window.top;
  } catch (e) {
    isLinkedIframe = true;
  }
  if (isLinkedIframe) {
    var iframeBanner = document.getElementById('pwaIframeBanner');
    if (iframeBanner) iframeBanner.classList.remove('hidden');
  } else {
    setTimeout(function() {
      var installBanner = document.getElementById('pwaInstallBanner');
      // If we got the beforeinstallprompt, show the banner, otherwise show for awareness
      if (installBanner) {
        installBanner.classList.remove('hidden');
      }
    }, 2000);
  }
}
