document.getElementById("eId").value=s.id;document.getElementById("eRegion").value=s.region;onEditRegionChange();

document.getElementById("eUpazila").value=s.upazila||"";
document.getElementById("eNursery").value=s.nurseryName;document.getElementById("eMobile").value=s.mobile;document.getElementById("eAddress").value=s.address||"";document.getElementById("eGeo").value=s.geoLocation||"";
document.getElementById("editSeedlingSections").innerHTML=buildSeedlingSections("editSeedlingSections",s);document.getElementById("editModal").classList.remove("hidden")},60);
};

});
